########começar um projeto no serveless framework#########

--se nao tiver o serverless instalado, instale com o comando abaixo no terminal
npm install -g serverless

--vá até a pasta desejada, para criar o projeto digite o comando abaixo e siga as instruções do terminal
serveless

--dentro da pasta do projeto criada crie uma virtual venv
pip install --user pipenv
OU
virtualenv venv --python=python3

--após instalado a virtual venv, entre dentro dela
source venv/bin/activate

--atualize o pip
python -m pip install --upgrade pip

--instale todas as bibliotecas desejadas e passe para o arquivo requirements.txt
pip freeze > requirements.txt

--biblioteca do psycopg2
python -m pip install psycopg2-binary

--caso esteja criando o projeto, instale o plugin
sls plugin install -n serverless-python-requirements

--caso não possua, instale o docker na sua maquina, se não instalar ele nao vai conseguir criar a imagem com as bibliotecas
necessárias para o script funcionar
https://www.docker.com/

--vai no arquivo serverless.yml e altera essa linha
custom:
  pythonRequirements:
    dockerizePip: true

--para subir o projeto, digite
serverless deploy
